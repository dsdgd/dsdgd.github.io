Seajs初始化配置

书写省略
当引入的模块文件时.js后缀时，后缀可省略不写。
/**未省略**/
seajs.use('./a/b/c/d.js', function (){...})

/**省略**/
seajs.use('./a/b/c/d', function (){...})
别名配置
用于use方法加载模块时更方便，给模块提供一个简短的命名。
seajs.config({
	alias : {
    	'name1' : './a/b/c/d.js'
    }
})

/**未使用别名时调用**/
seajs.use('./a/b/c/d.js', function (){...})

/**使用别名时调用**/
seajs.use('name1', function (){...})
路径配置
类似别名配置，用于加载同一路径下的多个文件。
seajs.config({
	paths : {
    	'name1' : './a/b/c'
    }
})

/**未使用路径配置时**/
seajs.use(['./a/b/c/a.js', './a/b/c/b.js'], function (){...})

/**使用路径配置时**/
seajs.use(['name1/a.js', 'name1/b.js'], function (){...})
变量配置
用于路径需要根据某些特定条件时来生成。
function paths(){
	return 'link';
}

seajs.config({
	vars : {
    	'locale' : paths()
    }
})

seajs.use(['{vars}/a.js', '{vars}/a.js'], function (){...})
