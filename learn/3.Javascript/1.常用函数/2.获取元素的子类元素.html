function Childs(obj){
	var result = [];
    var chl = obj.childNodes;
    var len = chl.length;
    
    for(var i = 0; i < len; i++){
    	chl[i].nodeType === 1 ? result.push(chl[i]) : false;
    };
    
    return result;
};
